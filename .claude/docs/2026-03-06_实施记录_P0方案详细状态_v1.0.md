        progress_backup = read_memory(f"progress-{project_id}")

        # Step 2: 保存 Checkpoint
        checkpoint_id = f"checkpoint-{project_id}-{phase}-{generate_uuid()}"
        write_memory(checkpoint_id, checkpoint_data)

        # Step 3: 更新 Progress
        progress_data = progress_backup.copy()
        progress_data.phases[phase].status = "completed"
        progress_data.phases[phase].end_time = get_current_timestamp()
        write_memory(f"progress-{project_id}", progress_data)

        return True

    except Exception as e:
        # ❌ 缺失：失败时恢复备份
        log_error(f"保存 Checkpoint 失败: {e}")
        if progress_backup:
            write_memory(f"progress-{project_id}", progress_backup)
        return False

def migrate_data_if_needed(data):
    """如果需要则迁移数据"""
    current_version = data.get("metadata", {}).get("version")

    if current_version == "1.0":
        # 备份原数据
        backup_data = data.copy()

        try:
            # 执行迁移 v1.0 → v2.0
            migrated_data = migrate_v1_to_v2(data)

            # 验证迁移结果
            if validate_data(migrated_data):
                return migrated_data
            else:
                # 验证失败，恢复备份
                return backup_data
        except Exception as e:
            # 迁移失败，恢复备份
            log_error(f"Migration failed: {e}")
            return backup_data

    return data

def migrate_v1_to_v2(data):
