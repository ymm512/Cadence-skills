```python
def validate_progress_data(progress_data):
    """
    验证 Progress 数据

    Args:
        progress_data: Progress 数据

    Returns:
        dict: 验证结果
    """
    errors = []

    # Step 1: 检查必填字段
    required_fields = [
        "metadata.version",
        "metadata.project_id",
        "metadata.project_name",
        "metadata.flow_type",
        "project_info.name",
        "project_info.current_phase",
        "overall_progress.percentage"
    ]

    for field_path in required_fields:
        if not has_field(progress_data, field_path):
            errors.append(f"缺少必填字段: {field_path}")

    # Step 2: 检查版本号格式
    if not re.match(r'^\d+\.\d+$', progress_data.metadata.version):
        errors.append("版本号格式错误,应为 X.Y 格式")

    # Step 3: 检查流程类型
    valid_flow_types = ["full-flow", "quick-flow", "exploration-flow"]
    if progress_data.metadata.flow_type not in valid_flow_types:
        errors.append(f"无效的流程类型: {progress_data.metadata.flow_type}")

    # Step 4: 检查进度百分比范围
    percentage = progress_data.overall_progress.percentage
    if not (0 <= percentage <= 100):
        errors.append(f"进度百分比超出范围: {percentage}")

    # Step 5: 检查阶段状态
    valid_statuses = ["completed", "in_progress", "pending"]
    for phase in progress_data.phases:
        if phase.status not in valid_statuses:
            errors.append(f"无效的阶段状态: {phase.status}")

    # 返回验证结果
    if errors:
        return {
            "valid": False,
            "errors": errors
        }
    else:
        return {
            "valid": True,
            "errors": []
        }
```

```python
def validate_checkpoint_data(checkpoint_data):
    """
    验证 Checkpoint 数据

    Args:
        checkpoint_data: Checkpoint 数据

    Returns:
        dict: 验证结果
    """
    errors = []

    # Step 1: 检查必填字段
    required_fields = [
        "metadata.version",
        "metadata.checkpoint_id",
        "metadata.project_id",
        "phase",
        "status",
        "timestamp"
    ]

    for field_path in required_fields:
        if not has_field(checkpoint_data, field_path):
            errors.append(f"缺少必填字段: {field_path}")

    # Step 2: 检查 Checkpoint ID 格式
    checkpoint_id = checkpoint_data.metadata.checkpoint_id
    if not is_valid_uuid(checkpoint_id):
        errors.append(f"无效的 Checkpoint ID: {checkpoint_id}")

    # Step 3: 检查阶段名称
    valid_phases = [
        "brainstorm", "analyze", "requirement", "design", "plan",
        "git-worktrees", "implementation", "test", "review"
    ]
    if checkpoint_data.phase not in valid_phases:
        errors.append(f"无效的阶段名称: {checkpoint_data.phase}")

    # Step 4: 检查状态
    valid_statuses = ["completed", "in_progress", "failed"]
    if checkpoint_data.status not in valid_statuses:
        errors.append(f"无效的状态: {checkpoint_data.status}")

    # Step 5: 检查时间戳格式
    if not is_valid_timestamp(checkpoint_data.timestamp):
        errors.append(f"无效的时间戳: {checkpoint_data.timestamp}")

    # 返回验证结果
    if errors:
        return {
            "valid": False,
            "errors": errors
        }
    else:
        return {
            "valid": True,
            "errors": []
        }
```

### 验证失败处理

```python
def handle_validation_failure(data_type, validation_result):
    """
    处理验证失败

    Args:
        data_type: 数据类型
        validation_result: 验证结果

    Returns:
        None
    """
    # 记录错误日志
    log_error(f"{data_type} 数据验证失败:")
    for error in validation_result["errors"]:
        log_error(f"  - {error}")

    # 拒绝写入
    raise ValidationError(f"{data_type} 数据验证失败", validation_result["errors"])
```

```python
def auto_cleanup():
    """
    自动清理过期数据

    Returns:
        dict: 清理统计
    """
    stats = {
        "archived": 0,
        "deleted": 0,
        "errors": 0
    }

    # Step 1: 扫描所有 memory
    memories = list_all_memories()

    # Step 2: 检查过期时间
    for memory_key in memories:
        memory_data = read_memory(memory_key)

        if not memory_data or "expires_at" not in memory_data:
            continue

        # 检查是否过期
        if is_expired(memory_data.expires_at):
            # Step 3: 归档或删除
            if should_archive(memory_data):
                if archive_memory(memory_key, memory_data):
                    stats["archived"] += 1
                else:
                    stats["errors"] += 1
            else:
                if delete_memory(memory_key):
                    stats["deleted"] += 1
                else:
                    stats["errors"] += 1

    # 返回清理统计
    return stats
```

```python
def archive_memory(memory_key, memory_data):
    """
    归档 memory

    Args:
        memory_key: Memory 键名
        memory_data: Memory 数据

    Returns:
        bool: 是否成功归档
    """
    try:
        # Step 1: 压缩数据
        compressed_data = compress_data(memory_data)

        # Step 2: 生成归档键名
        archive_key = generate_archive_key(memory_key)

        # Step 3: 移动到归档
        write_memory(archive_key, compressed_data)

        # Step 4: 删除原数据
        delete_memory(memory_key)

        # Step 5: 记录归档日志
        log_info(f"已归档: {memory_key} → {archive_key}")

        return True

    except Exception as e:
        log_error(f"归档失败: {memory_key}, 错误: {e}")
        return False
```

```python
def cleanup_command(dry_run=False, archive_only=False, delete_only=False):
    """
    执行清理命令

    Args:
        dry_run: 是否只预览不执行
        archive_only: 是否只归档不删除
        delete_only: 是否只删除不归档

    Returns:
        None
    """
    print("🧹 开始清理过期数据...\n")

    if dry_run:
        print("⚠️  预览模式 - 不会实际执行清理\n")

    # 扫描过期数据
    expired_memories = scan_expired_memories()

    if not expired_memories:
        print("✅ 没有过期数据需要清理")
        return

    # 分类统计
    to_archive = []
    to_delete = []

    for memory in expired_memories:
        if should_archive(memory):
            to_archive.append(memory)
        else:
            to_delete.append(memory)

    # 显示统计
    print(f"📊 清理统计:")
    print(f"  - 待归档: {len(to_archive)} 条")
    print(f"  - 待删除: {len(to_delete)} 条\n")

    if dry_run:
        # 显示详细信息
        if to_archive:
            print("待归档:")
            for memory in to_archive[:10]:  # 只显示前10条
                print(f"  - {memory.key}")
            if len(to_archive) > 10:
                print(f"  ... 还有 {len(to_archive) - 10} 条\n")

        if to_delete:
            print("待删除:")
            for memory in to_delete[:10]:  # 只显示前10条
                print(f"  - {memory.key}")
            if len(to_delete) > 10:
                print(f"  ... 还有 {len(to_delete) - 10} 条\n")

        return

    # 执行归档
    if not delete_only and to_archive:
        print("📦 正在归档...")
        archived_count = 0
        for memory in to_archive:
            if archive_memory(memory.key, memory.data):
                archived_count += 1
        print(f"✅ 已归档: {archived_count}/{len(to_archive)} 条\n")

    # 执行删除
    if not archive_only and to_delete:
        print("🗑️  正在删除...")
        deleted_count = 0
        for memory in to_delete:
            if delete_memory(memory.key):
                deleted_count += 1
        print(f"✅ 已删除: {deleted_count}/{len(to_delete)} 条\n")

    print("✨ 清理完成!")
```

```yaml
metadata:
```

### 版本兼容性策略

#### 版本号规则

```yaml
版本号格式: X.Y

X: 主版本号(不兼容的更改)
Y: 次版本号(向后兼容的更改)

示例:
- 1.0 → 1.1: 新增字段(向后兼容)
- 1.0 → 2.0: 重命名字段(不兼容)
```

---

#### 兼容性类型

```yaml
兼容性类型:
  - 新增字段(向后兼容):
      示例: v1.0 → v1.1
      处理: 新字段设置默认值
      迁移: 自动

  - 重命名字段(需迁移):
      示例: v1.0 → v2.0
      处理: 数据迁移脚本
      迁移: 手动

  - 删除字段(需迁移):
      示例: v1.0 → v2.0
      处理: 数据迁移脚本
      迁移: 手动
```

### 迁移流程

#### 检测版本

```python
def detect_data_version(data):
    """
    检测数据版本

    Args:
        data: 数据对象

    Returns:
        str: 版本号
    """
    if "metadata" in data and "version" in data["metadata"]:
        return data["metadata"]["version"]
    else:
        return "1.0"  # 默认版本
```

#### 执行迁移

```python
def migrate_data_if_needed(data, target_version="1.1"):
    """
    如果需要,执行数据迁移

    Args:
        data: 数据对象
        target_version: 目标版本

    Returns:
        dict: 迁移后的数据
    """
    current_version = detect_data_version(data)

    # 如果已是目标版本,无需迁移
    if current_version == target_version:
        return data

    # Step 1: 备份原数据
    backup_data = deepcopy(data)

    try:
        # Step 2: 执行迁移脚本
        if current_version == "1.0" and target_version == "1.1":
            migrated_data = migrate_v1_0_to_v1_1(data)
        elif current_version == "1.1" and target_version == "2.0":
            migrated_data = migrate_v1_1_to_v2_0(data)
        else:
            raise MigrationError(f"不支持的版本迁移: {current_version} → {target_version}")

        # Step 3: 验证迁移结果
        if not validate_migrated_data(migrated_data):
            raise MigrationError("迁移后数据验证失败")

        # Step 4: 返回迁移后的数据
        return migrated_data

    except Exception as e:
        # Step 5: 失败时恢复备份
        log_error(f"数据迁移失败: {e}")
        return backup_data
```

#### 迁移脚本示例

```python
def migrate_v1_0_to_v1_1(data):
    """
    从 v1.0 迁移到 v1.1

    变更:
    - 新增 metadata.flow_type 字段
    - 新增 time_stats.estimated_remaining 字段

    Args:
        data: v1.0 数据

    Returns:
        dict: v1.1 数据
    """
    # Step 1: 更新版本号
    data["metadata"]["version"] = "1.1"

    # Step 2: 添加新字段(设置默认值)
    if "flow_type" not in data["metadata"]:
        data["metadata"]["flow_type"] = "full-flow"  # 默认值

    if "time_stats" not in data:
        data["time_stats"] = {}

    if "estimated_remaining" not in data["time_stats"]:
        data["time_stats"]["estimated_remaining"] = 0  # 默认值

    return data


def migrate_v1_1_to_v2_0(data):
    """
    从 v1.1 迁移到 v2.0

    变更:
    - 重命名 project_info.name → project_info.project_name
    - 删除 deprecated_field

    Args:
        data: v1.1 数据

    Returns:
        dict: v2.0 数据
    """
    # Step 1: 更新版本号
    data["metadata"]["version"] = "2.0"

    # Step 2: 重命名字段
    if "project_info" in data and "name" in data["project_info"]:
        data["project_info"]["project_name"] = data["project_info"]["name"]
        del data["project_info"]["name"]

    # Step 3: 删除废弃字段
    if "deprecated_field" in data:
        del data["deprecated_field"]

    return data
```

```python
def validate_migrated_data(data):
    """
    验证迁移后的数据

    Args:
        data: 迁移后的数据

    Returns:
        bool: 是否验证通过

        return False

    # Step 2: 验证必填字段
    if not has_required_fields(data):
        return False

    # Step 3: 验证数据完整性
```
