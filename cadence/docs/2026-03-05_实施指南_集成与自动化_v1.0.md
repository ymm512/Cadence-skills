## 📊 进度自动计算

### 计算公式

#### 整体进度百分比

```python
def calculate_overall_progress(progress_data):
    """
    计算整体进度百分比

    Args:
        progress_data: Progress 数据

    Returns:
        float: 进度百分比(0-100)
    """
    total_phases = len(progress_data.phases)
    completed_phases = sum(
        1 for phase in progress_data.phases
        if phase.status == "completed"
    )

    percentage = (completed_phases / total_phases) * 100

    return percentage
```

#### 时间统计

```python
def calculate_time_stats(progress_data):
    """
    计算时间统计

    Args:
        progress_data: Progress 数据

    Returns:
        dict: 时间统计数据
    """
    # 计算已用时间
    total_time = 0
    for phase in progress_data.phases:
        if phase.status == "completed":
            if phase.start_time and phase.end_time:
                total_time += (phase.end_time - phase.start_time)

    # 计算预估剩余时间
    completed_phases = sum(
        1 for phase in progress_data.phases
        if phase.status == "completed"
    )

    if completed_phases > 0:
        avg_time_per_phase = total_time / completed_phases
        remaining_phases = len(progress_data.phases) - completed_phases
        estimated_remaining = avg_time_per_phase * remaining_phases
    else:
        estimated_remaining = 0

    return {
        "total_time": total_time,
        "estimated_remaining": estimated_remaining
    }
```

---

### 自动计算触发时机

```yaml
触发时机:
  - 节点开始时:
      - 记录 start_time
      - 更新阶段状态为 "in_progress"

  - 节点结束时:
      - 记录 end_time
      - 更新阶段状态为 "completed"
      - 重新计算进度
      - 重新计算时间统计

  - 查询进度时:
      - 实时计算进度百分比
      - 实时计算时间统计
```

### 完整示例

```python
def update_progress_on_phase_completion(project_id, phase_name):
    """
    节点完成时更新进度

    Args:
        project_id: 项目ID
        phase_name: 阶段名称

    Returns:
        dict: 更新后的进度数据
    """
    # Step 1: 读取 Progress
    progress_data = read_memory(f"progress-{project_id}")

    # Step 2: 更新阶段状态
    for phase in progress_data.phases:
        if phase.phase_name == phase_name:
            phase.status = "completed"
            phase.end_time = get_current_timestamp()
            break

    # Step 3: 计算整体进度
    progress_data.overall_progress.percentage = calculate_overall_progress(progress_data)
    progress_data.overall_progress.completed_phases = sum(
        1 for phase in progress_data.phases
        if phase.status == "completed"
    )

    # Step 4: 计算时间统计
    time_stats = calculate_time_stats(progress_data)
    progress_data.time_stats.total_time = time_stats["total_time"]
    progress_data.time_stats.estimated_remaining = time_stats["estimated_remaining"]

    # Step 5: 更新时间戳
    progress_data.metadata.updated_at = get_current_timestamp()

    # Step 6: 保存 Progress
    write_memory(f"progress-{project_id}", progress_data)

    return progress_data
```

```python
def detect_interrupted_sessions():
    """
    检测中断的会话

    Returns:
        list: 未完成的会话列表
    """
    # Step 1: 列出所有 progress 记录
    progress_keys = list_memories("progress-*")

    # Step 2: 过滤未完成的记录
    interrupted_sessions = []

    for key in progress_keys:
        progress_data = read_memory(key)

        # 检查是否未完成
        if progress_data.overall_progress.percentage < 100:
            interrupted_sessions.append({
                "project_id": progress_data.metadata.project_id,
                "project_name": progress_data.metadata.project_name,
                "flow_type": progress_data.metadata.flow_type,
                "percentage": progress_data.overall_progress.percentage,
                "completed_phases": progress_data.overall_progress.completed_phases,
                "total_phases": progress_data.overall_progress.total_phases,
                "last_phase": get_last_completed_phase(progress_data),
                "interrupted_at": progress_data.metadata.updated_at,
                "git_branch": progress_data.project_info.git_branch
            })

    return interrupted_sessions
```

```python
def select_session_for_resume():
    """
    用户选择要恢复的会话

    Returns:
        dict: 选中的会话
    """
    sessions = detect_interrupted_sessions()

    if not sessions:
        print("✅ 没有找到可恢复的会话")
        return None

    list_resumable_sessions()

    choice = input("选择要恢复的会话 [1-{}]: ".format(len(sessions)))

    try:
        index = int(choice) - 1
        if 0 <= index < len(sessions):
            return sessions[index]
        else:
            print("❌ 无效的选择")
            return None
    except ValueError:
        print("❌ 请输入数字")
        return None
```

#### 4. 重建上下文

**重建步骤**:

```python
def rebuild_context(project_id):
    """
    重建项目上下文

    Args:
        project_id: 项目ID

    Returns:
        bool: 是否成功重建
    """
    # Step 1: 读取 Progress
    progress_data = read_memory(f"progress-{project_id}")

    # Step 2: 确定下一个待执行的节点
    next_phase = None
    for phase in progress_data.phases:
        if phase.status == "pending":
            next_phase = phase.phase_name
            break

    if not next_phase:
        print("✅ 项目已完成,无需恢复")
        return False

    # Step 3: 读取最近的 Checkpoint
    checkpoint_id = find_latest_checkpoint(project_id, next_phase)
    if checkpoint_id:
        checkpoint_data = read_memory(checkpoint_id)

        # Step 4: 恢复 Git 分支
        git_branch = checkpoint_data.context.git_branch
        checkout_git_branch(git_branch)

        # Step 5: 恢复 TodoWrite 状态
        todowrite_state = checkpoint_data.context.todowrite_state
        restore_todowrite_state(todowrite_state)

        # Step 6: 加载项目上下文
        project_context = checkpoint_data.context.project_context
        load_project_context(project_context)

    # Step 7: 显示当前状态
    display_current_status(progress_data, next_phase)

    return True
```

```python
def resume_execution(project_id):
    """
    恢复执行

    Args:
        project_id: 项目ID

    Returns:
        None
    """
    # Step 1: 重建上下文
    if not rebuild_context(project_id):
        return

    # Step 2: 读取 Progress
    progress_data = read_memory(f"progress-{project_id}")

    # Step 3: 找到下一个待执行的节点
    next_phase = None
    for phase in progress_data.phases:
        if phase.status == "pending":
            next_phase = phase.phase_name
            break

    # Step 4: 开始执行
    print(f"\n▶️  开始执行: {next_phase}\n")

    # 根据流程类型调用对应的 skill
    if progress_data.metadata.flow_type == "full-flow":
        execute_full_flow_phase(next_phase)
    elif progress_data.metadata.flow_type == "quick-flow":
        execute_quick_flow_phase(next_phase)
    elif progress_data.metadata.flow_type == "exploration-flow":
        execute_exploration_flow_phase(next_phase)
```

```python
def resume_session():
    """
    完整的会话恢复流程

    Returns:
        None
    """
    print("🔄 恢复会话\n")

    # Step 1: 检测中断场景
    sessions = detect_interrupted_sessions()

    if not sessions:
        print("✅ 没有找到可恢复的会话")
        return

    # Step 2: 列出可恢复会话
    list_resumable_sessions()

    # Step 3: 用户选择恢复点
    selected_session = select_session_for_resume()

    if not selected_session:
        return

    # Step 4: 重建上下文
    project_id = selected_session["project_id"]

    print(f"\n🔄 恢复项目: {selected_session['project_name']}\n")

    if not rebuild_context(project_id):
        return

    # Step 5: 继续执行
    resume_execution(project_id)
```
