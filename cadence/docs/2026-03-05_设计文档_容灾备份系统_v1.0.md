#### 自动备份触发

```python
def auto_backup_if_needed():
    """
    如果需要,执行自动备份

    Returns:
        bool: 是否执行了备份
    """
    # Step 1: 检查是否启用自动备份
    config = load_backup_config()

    if not config["enabled"]:
        return False

    # Step 2: 检查上次备份时间
    last_backup = get_last_backup_time()
    current_time = get_current_timestamp()

    if last_backup:
        time_since_last_backup = current_time - last_backup

        # 如果距离上次备份不足 interval,跳过
        if time_since_last_backup < config["interval_seconds"]:
            return False

    # Step 3: 执行备份
    create_backup(backup_type="auto")

    return True
```

```python
def create_incremental_backup():
    """
    创建增量备份

    Returns:
        str: 备份ID
    """
    # Step 1: 找到最近的全量备份
    last_full_backup = find_last_full_backup()

    if not last_full_backup:
        # 如果没有全量备份,创建全量备份
        return create_full_backup()

    # Step 2: 计算差异
    current_data = export_all_data()
    baseline_data = load_backup_data(last_full_backup)

    delta = calculate_delta(baseline_data, current_data)

    # Step 3: 保存增量备份
    backup_id = generate_backup_id()
    backup_data = {
        "metadata": {
            "backup_id": backup_id,
            "backup_type": "incremental",
            "base_backup_id": last_full_backup["metadata"]["backup_id"],
            "created_at": get_current_timestamp()
        },
        "delta": delta
    }

    save_backup(backup_id, backup_data)

    return backup_id
```

```python
def create_full_backup():
    """
    创建全量备份

    Returns:
        str: 备份ID
    """
    # Step 1: 导出所有数据
    all_data = export_all_data()

    # Step 2: 生成备份ID
    backup_id = generate_backup_id()

    # Step 3: 创建备份数据
    backup_data = {
        "metadata": {
            "backup_id": backup_id,
            "backup_type": "full",
            "created_at": get_current_timestamp(),
            "data_count": count_data_items(all_data),
            "size_bytes": calculate_size(all_data)
        },
        "data": all_data
    }

    # Step 4: 压缩备份数据
    compressed_data = compress_data(backup_data)

    # Step 5: 保存备份
    save_backup(backup_id, compressed_data)

    return backup_id


def export_all_data():
    """
    导出所有数据

    Returns:
        dict: 所有数据
    """
    all_data = {}

    # Step 1: 导出 Progress 数据
    progress_keys = list_memories("progress-*")
    for key in progress_keys:
        all_data[key] = read_memory(key)

    # Step 2: 导出 Checkpoint 数据
    checkpoint_keys = list_memories("checkpoint-*")
    for key in checkpoint_keys:
        all_data[key] = read_memory(key)

    # Step 3: 导出 Session Summary 数据
    session_keys = list_memories("session-*")
    for key in session_keys:
        all_data[key] = read_memory(key)

    # Step 4: 导出索引数据
    index_keys = list_memories("index-*")
    for key in index_keys:
        all_data[key] = read_memory(key)

    return all_data
```

#### 恢复实现

```python
def restore_backup(backup_id, dry_run=False):
    """
    恢复备份

    Args:
        backup_id: 备份ID
        dry_run: 是否只预览不执行

    Returns:
        bool: 是否成功恢复
    """
    print(f"🔄 恢复备份: {backup_id}\n")

    if dry_run:
        print("⚠️  预览模式 - 不会实际恢复数据\n")

    # Step 1: 读取备份数据
    backup_data = load_backup(backup_id)

    if not backup_data:
        print(f"❌ 备份不存在: {backup_id}")
        return False

    # Step 2: 验证备份完整性
    if not validate_backup(backup_data):
        print("❌ 备份验证失败")
        return False

    # Step 3: 显示备份信息
    display_backup_info(backup_data)

    if dry_run:
        return True

    # Step 4: 备份当前数据
    print("📦 备份当前数据...")
    current_backup_id = create_backup(backup_type="pre-restore")
    print(f"✅ 已备份: {current_backup_id}\n")

    # Step 5: 恢复数据
    print("📥 恢复数据...")

    if backup_data["metadata"]["backup_type"] == "full":
        restore_full_backup(backup_data)
    elif backup_data["metadata"]["backup_type"] == "incremental":
        restore_incremental_backup(backup_data)

    # Step 6: 验证恢复结果
    print("🔍 验证恢复结果...")
    if not verify_restored_data():
        print("❌ 恢复验证失败,回滚到恢复前状态")
        restore_backup(current_backup_id)
        return False

    print("✅ 恢复成功!")
    return True
```

```python
def restore_to_point_in_time(target_time, dry_run=False):
    """
    恢复到指定时间点

    Args:
        target_time: 目标时间
        dry_run: 是否只预览不执行

    Returns:
        bool: 是否成功恢复
    """
    print(f"🔄 恢复到时间点: {target_time}\n")

    # Step 1: 列出目标时间之前的所有备份
    backups = list_backups_before_time(target_time)

    if not backups:
        print(f"❌ 没有找到 {target_time} 之前的备份")
        return False

    # Step 2: 找到最近的全量备份
    full_backup = find_nearest_full_backup(backups, target_time)

    if not full_backup:
        print("❌ 没有找到全量备份")
        return False

    print(f"📋 找到全量备份: {full_backup['metadata']['backup_id']}")

    # Step 3: 找到全量备份之后的所有增量备份
    incremental_backups = find_incremental_backups_between(
        full_backup["metadata"]["created_at"],
        target_time
    )

    print(f"📋 找到 {len(incremental_backups)} 个增量备份\n")

    # Step 4: 恢复全量备份
    if not restore_backup(full_backup["metadata"]["backup_id"], dry_run):
        return False

    # Step 5: 依次应用增量备份
    for backup in incremental_backups:
        print(f"📥 应用增量备份: {backup['metadata']['backup_id']}")
        apply_incremental_backup(backup)

    print(f"\n✅ 已恢复到 {target_time}")
    return True
```

```python
def backup_list_command(backup_type=None, limit=10):
    """
    列出备份

    Args:
        backup_type: 备份类型(可选)
        limit: 显示数量限制

    Returns:
        None
    """
    print("📋 备份列表:\n")

    # Step 1: 列出所有备份
    backups = list_all_backups()

    # Step 2: 过滤备份类型
    if backup_type:
        backups = [b for b in backups if b["metadata"]["backup_type"] == backup_type]

    # Step 3: 按时间排序
    backups.sort(key=lambda b: b["metadata"]["created_at"], reverse=True)

    # Step 4: 限制数量
    backups = backups[:limit]

    # Step 5: 显示备份列表
    if not backups:
        print("📭 没有找到备份")
        return

    for backup in backups:
        print(f"- **{backup['metadata']['backup_id']}**")
        print(f"  - 类型: {backup['metadata']['backup_type']}")
        print(f"  - 创建时间: {format_timestamp(backup['metadata']['created_at'])}")

        if "name" in backup["metadata"]:
            print(f"  - 名称: {backup['metadata']['name']}")

        if "data_count" in backup["metadata"]:
            print(f"  - 数据量: {backup['metadata']['data_count']} 条")

        if "size_bytes" in backup["metadata"]:
            print(f"  - 大小: {format_size(backup['metadata']['size_bytes'])}")

        print()
```

```python
def backup_delete_command(backup_id):
    """
    删除备份

    Args:
        backup_id: 备份ID

    Returns:
        None
    """
    print(f"🗑️  删除备份: {backup_id}\n")

    # Step 1: 检查备份是否存在
    backup_data = load_backup(backup_id)

    if not backup_data:
        print(f"❌ 备份不存在: {backup_id}")
        return

    # Step 2: 显示备份信息
    display_backup_info(backup_data)

    # Step 3: 确认删除
    confirm = input("确认删除此备份? [y/N]: ")

    if confirm.lower() != 'y':
        print("❌ 取消删除")
        return

    # Step 4: 删除备份
    delete_backup(backup_id)

    print(f"\n✅ 备份已删除: {backup_id}")
```

### 备份数据模型

```yaml
Backup:
  # 元数据
  metadata:
    backup_id: string                # 备份ID
    backup_type: "full" | "incremental"  # 备份类型
    name: string | null              # 备份名称(可选)
    created_at: timestamp            # 创建时间
    base_backup_id: string | null    # 基准备份ID(增量备份)

    # 统计信息
    data_count: number               # 数据条数
    size_bytes: number               # 备份大小(字节)

    # 校验信息
    checksum: string                 # 校验和
    checksum_algorithm: string       # 校验算法

  # 数据(全量备份)
  data: object | null                # 完整数据

  # 差异(增量备份)
  delta: object | null               # 差异数据
```
